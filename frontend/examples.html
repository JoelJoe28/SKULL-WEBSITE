🦴 SKULL API Integration Examples

1. Login Example

async function loginExample() { try { const result = await api.login('user@example.com', 'password123'); console.log('Login successful:', result); // Token is automatically stored in localStorage // User data is available in result.user } catch (error) { console.error('Login failed:', api.handleError(error)); } }

2. Get User Profile Example

async function getProfileExample() { try { const user = await api.getProfile(); console.log('User profile:', user); // User data includes: id, email, name, phone, etc. } catch (error) { console.error('Profile fetch failed:', api.handleError(error)); } }

3. Get Orders Example

async function getOrdersExample() { try { const orders = await api.getOrders(); console.log('User orders:', orders); // Orders array with order details, items, status, etc. } catch (error) { console.error('Orders fetch failed:', api.handleError(error)); } }

4. Create Order Example

async function createOrderExample() { try { const orderData = { items: [ { id: 1, name: 'Skull T-Shirt', price: 599, quantity: 2, category: 'TOPS' } ], totalAmount: 1198, paymentMethod: 'COD' }; const order = await api.createOrder(orderData); console.log('Order created:', order); // Returns order confirmation with ID and status } catch (error) { console.error('Order creation failed:', api.handleError(error)); } }

5. Update Profile Example

async function updateProfileExample() { try { const updateData = { name: 'John Doe', phone: '+1234567890', firstName: 'John', lastName: 'Doe' }; const updatedUser = await api.updateProfile(updateData); console.log('Profile updated:', updatedUser); // Returns updated user data } catch (error) { console.error('Profile update failed:', api.handleError(error)); } }
